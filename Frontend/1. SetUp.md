# 1. Env Variables In Vite

Vite exposes env variables under import.meta.env object as strings automatically.

To prevent accidentally leaking env variables to the client, only variables prefixed with VITE\_ are exposed to your Vite-processed code. e.g. for the following env variables:

.env

```js
VITE_SOME_KEY = 123;
DB_PASSWORD = foobar;
```

Only `VITE_SOME_KEY` will be exposed as `import.meta.env.VITE_SOME_KEY` to your client source code, but `DB_PASSWORD` will not.

```js
console.log(import.meta.env.VITE_SOME_KEY); // "123"
console.log(import.meta.env.DB_PASSWORD); // undefined
```

## Desktop Category Navigation

`CategoryDropdown.jsx`

```js
import {
  useState,
  useRef,
  useEffect,
  useLayoutEffect,
  useCallback,
} from "react";
import { FaChevronDown, FaChevronRight } from "react-icons/fa";
import { Link } from "react-router-dom";
import { categories } from "../../utils/data";

const CategoryDropdown = () => {
  const [isOpen, setIsOpen] = useState(false);
  const [hoveredIndex, setHoveredIndex] = useState(null);
  const dropdownRef = useRef(null);

  // close on outside click
  useEffect(() => {
    const handleClickOutside = (e) => {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target)) {
        setIsOpen(false);
        setHoveredIndex(null);
      }
    };
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  return (
    <div className="relative" ref={dropdownRef}>
      {/* Button */}
      <button
        onClick={() => setIsOpen((s) => !s)}
        className="flex items-center gap-1 text-[12px] border border-secondary  text-secondary px-2 py-1 rounded-md cursor-pointer"
      >
        Categories <FaChevronDown size={10} className="text-sm mt-1" />
      </button>

      {/* Main Dropdown */}
      {isOpen && (
        <div className="absolute left-0 mt-3 pb-4 w-[300px]  bg-white shadow-[0px_4px_6px_0px_rgba(0,_0,_0,_0.1)] z-50">
          <ul className="flex flex-col text-sm relative">
            {categories?.map((category, index) => (
              <CategoryItem
                key={index}
                category={category}
                isHovered={hoveredIndex === index}
                onEnter={() => setHoveredIndex(index)}
                onLeave={() => setHoveredIndex(null)}
                onSelect={() => setIsOpen(false)}
              />
            ))}
          </ul>
        </div>
      )}
    </div>
  );
};

const CategoryItem = ({ category, isHovered, onEnter, onLeave, onSelect }) => {
  const itemRef = useRef(null);
  return (
    <li
      ref={itemRef}
      onMouseEnter={onEnter}
      onMouseLeave={onLeave}
      className="group flex items-center  justify-between border-b-[1px] border-border-light px-4 py-2
       hover:bg-gray-100 transition cursor-pointer relative"
    >
      <div className="flex items-center gap-4">
        <img
          src={category.image}
          alt={category.name}
          className="rounded-full w-7 h-7"
        />
        {category.name}
      </div>
      <FaChevronRight className="text-xs text-gray-400 group-hover:text-gray-700" />

      {isHovered && category?.items?.length > 0 && (
        <SmartSubmenu anchorRef={itemRef}>
          {category.items.map((sub, i) => (
            <li
              key={i}
              className="px-4 py-2 hover:bg-gray-100 transition border-b-[1px] border-border-light"
            >
              <Link
                onClick={onSelect}
                to={`/category${category.url
                  .toLowerCase()
                  .replace(/ /g, "-")}/${sub.toLowerCase().replace(/ /g, "-")}`}
              >
                {sub}
              </Link>
            </li>
          ))}
        </SmartSubmenu>
      )}
    </li>
  );
};

/**
 * SmartSubmenu
 * - Positions relative to the anchor <li>
 * - Prevents viewport overflow vertically (clamps top & bottom)
 * - Flips left when there isn't enough space on the right
 * - Adds max height with scroll
 */
const SmartSubmenu = ({ anchorRef, children }) => {
  const submenuRef = useRef(null);
  const [style, setStyle] = useState({}); // { top, left/right }
  const MARGIN = 8;
  const MAX_VH = 0.72; // 72vh cap for submenu

  const computePosition = useCallback(() => {
    const anchor = anchorRef.current;
    const menu = submenuRef.current;
    if (!anchor || !menu) return;

    // reset to measure natural size
    menu.style.maxHeight = "";
    menu.style.left = "100%";
    menu.style.right = "auto";
    menu.style.top = "0px";

    const anchorRect = anchor.getBoundingClientRect();

    // Ensure we know the menu‚Äôs natural height
    const maxHeightPx = Math.floor(window.innerHeight * MAX_VH);
    const naturalHeight = Math.min(menu.scrollHeight, maxHeightPx);

    // Horizontal placement: prefer right; flip left if not enough space
    const menuWidth = menu.offsetWidth || 224; // fallback if not rendered width yet
    const spaceRight = window.innerWidth - anchorRect.right;
    const spaceLeft = anchorRect.left;

    let openLeft = false;
    if (spaceRight < menuWidth + MARGIN && spaceLeft > spaceRight) {
      openLeft = true;
    }

    // Vertical placement: clamp inside viewport
    const spaceBelow = window.innerHeight - anchorRect.top; // from anchor top to bottom of viewport
    let topOffset = 0;

    if (spaceBelow < naturalHeight + MARGIN) {
      // shift upward so bottom fits
      topOffset = -(naturalHeight - spaceBelow + MARGIN);
      // but don't go above the viewport top
      if (anchorRect.top + topOffset < MARGIN) {
        topOffset = MARGIN - anchorRect.top;
      }
    }

    setStyle({
      top: `${topOffset}px`,
      ...(openLeft
        ? { right: "100%", left: "auto" }
        : { left: "100%", right: "auto" }),
      maxHeight: `${maxHeightPx}px`,
    });
  }, [anchorRef]);

  // Recompute on mount/hover, resize, and scroll
  useLayoutEffect(() => {
    computePosition();
    const onWinChange = () => computePosition();
    window.addEventListener("resize", onWinChange);
    window.addEventListener("scroll", onWinChange, true); // true => capture scroll from ancestors
    return () => {
      window.removeEventListener("resize", onWinChange);
      window.removeEventListener("scroll", onWinChange, true);
    };
  }, [computePosition]);

  return (
    <ul
      ref={submenuRef}
      className="absolute ml-1 w-56 bg-white shadow-[0px_4px_6px_0px_rgba(0,_0,_0,_0.1)] overflow-y-auto"
      style={style}
    >
      {children}
    </ul>
  );
};

export default CategoryDropdown;
```
