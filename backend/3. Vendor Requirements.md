# Category Dropdown

```js
import { useState, useRef, useEffect } from "react";
import { FaChevronDown, FaChevronRight } from "react-icons/fa";
import { Link } from "react-router-dom";

// import local file
import { categories } from "../../utils/data";

const CategoryDropdown = () => {
  const [isOpen, setIsOpen] = useState(false);
  const [hoveredCategory, setHoveredCategory] = useState(null);
  const dropdownRef = useRef();

  useEffect(() => {
    const handleClickOutside = (e) => {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target)) {
        setIsOpen(false);
        setHoveredCategory(null);
      }
    };
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  return (
    <div className="relative" ref={dropdownRef}>
      {/* Button */}
      <button
        onClick={() => setIsOpen(!isOpen)}
        // className="flex items-center gap-1 text-gray-800 font-medium hover:text-purple-600 cursor-pointer"
        className="flex items-center gap-1 bg-purple-600 hover:bg-purple-700 text-white px-4 pt-1 pb-2 rounded-md cursor-pointer"
      >
        Categories <FaChevronDown className="text-sm mt-1" />
      </button>

      {/* Main Dropdown */}
      {isOpen && (
        <div className="absolute left-0 mt-3 w-64 bg-white  shadow-[0px_4px_6px_0px_rgba(0,_0,_0,_0.1)] z-50">
          <ul className="flex flex-col text-sm relative">
            {/* {Object.keys(categoryData).map((cat, index) => ( */}
            {categories?.map((category, index) => (
              <li
                key={index}
                onMouseEnter={() => setHoveredCategory(category)}
                onMouseLeave={() => setHoveredCategory(null)}
                className="group flex items-center justify-between px-4 py-2 hover:bg-gray-100 transition cursor-pointer relative"
              >
                <span>{category.name}</span>
                <FaChevronRight className="text-xs text-gray-400 group-hover:text-gray-700" />

                {/* Submenu */}
                {hoveredCategory === category && categories?.length > 0 && (
                  <ul className="absolute top-0 left-full ml-1 w-52 bg-white border shadow-md">
                    {category.items.map((sub, i) => (
                      <li
                        key={i}
                        className="px-4 py-2 hover:bg-gray-100 transition"
                      >
                        <Link
                          to={`/category${category.url
                            .toLowerCase()
                            .replace(/ /g, "-")}/${sub
                            .toLowerCase()
                            .replace(/ /g, "-")}`}
                        >
                          {sub}
                        </Link>
                      </li>
                    ))}
                  </ul>
                )}
              </li>
            ))}
          </ul>
        </div>
      )}
    </div>
  );
};

export default CategoryDropdown;
```
